SELECT ID_I_PROF_ESPE,
       NOMBRE_PROFESIONAL,
       NOMBRE_ESPECIALIDAD,
       ID_M_PACIENTES,
       NOMBRE_COMPLETO,
       EDAD_UNIDAD,
       FECHA_EVOLUCION,
       ID_M_USUARIOS,
       NOMBRE_USUARIO,
       NOMBRE_SERVICIO,
       TIPO,
       CANCELADO 
FROM V_PACIENTES_POR_VER_CONSULTA 
WHERE ID_I_PROF_ESPE='0031358' AND FECHA_EVOLUCION BETWEEN TRUNC(SYSDATE) AND TRUNC(SYSDATE) + 1 ORDER BY FECHA_EVOLUCION
